Let \(x,y \in \X\).
\begin{enumerate}
    \item The \(1 \times 1\) kernel matrix \([k(x,x)]\) is positive semidefinite.
    So, \(k(x,x) \geq 0\).
    \item The \(2 \times 2\) kernel matrix
    \begin{equation}
        K = \begin{bmatrix}
            k(x,x) & k(x,y) \\ k(y,x) & k(y,y)
        \end{bmatrix}
    \end{equation}
    is positive semidefinite.
    Let \(v = \begin{bmatrix}
        k(y,y) \\ -k(x,y)
    \end{bmatrix}\).
    Then
    \begin{align}
        0 &\leq v^\top K v\\
        &= \begin{bmatrix}
            k(y,y) \\ -k(x,y)
        \end{bmatrix}^\top
        \begin{bmatrix}
            k(x,x)k(y,y) - k(x,y)^2 \\ 0
        \end{bmatrix}
        \notag\\
        % &= k(y,y) \left[
        %     k(y,y) k(x,x) - k(x,y)^2
        % \right]
        % \notag\\
        % &\qquad - k(x,y) \left[
        %     k(y,y) k(y,x) - k(x,y) k(y,y)
        %     \right]
        % \notag\\
        &= k(y,y) \left[
            k(x,x)k(y,y) - k(x,y)^2
        \right].
        \notag
    \end{align}
    Then \(v^\top K v \geq 0\) implies \(k(x,y)^2 \leq k(x,x) k(y,y)\).
    \qedhere
\end{enumerate}
