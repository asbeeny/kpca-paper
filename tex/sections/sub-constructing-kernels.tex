\begin{theorem}[Gaussian kernel]
    \label{thm:gaussian-kernel}
    The function \(k : \RR^n \times \RR^n \to \RR\) defined by
    \begin{equation}
        \label{eqn:gaussian-kernel}
        k(x,y) = \exp\left(\frac{-\|x-y\|^2}{2\sigma^2}\right),
    \end{equation}
    is a kernel.
